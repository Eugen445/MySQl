## 常见函数

###### 单行函数

```mysql
1.字符函数
	1).length获取参数的字节个数
		SELECT LENGTH('mike');
		SELECT LENGTH('张三');
		SHOW VARIABLES LIKE '%char%';
	2).concat拼接字符串
		SELECT 
  		CONCAT(last_name, '_', first_name) 姓名 
		FROM
 	 	employees ;
 	3).upper,lower
		SELECT 
  		CONCAT(UPPER(last_name),'_', LOWER(first_name)) 姓名 
		FROM
  		employees ;
  	4)substr, substring
		注意:索引从1开始
		SELECT SUBSTR('李莫愁爱上了陆展元', 7) out_put;
		SELECT SUBSTR('李莫愁爱上了陆展元', 1, 3) out_put;
		SELECT 
  			CONCAT(
    			UPPER(SUBSTR(last_name, 1, 1)),
    			'_',
    			LOWER(SUBSTR(last_name, 2)),
    			LOWER(first_name)
      		) AS 姓名
	5)instr 返回子串第一次出现的索引,如果找不到返回0
		SELECT 
  			INSTR(
    			'杨不梅爱上了殷六侠',
    			'殷六侠'
  			) AS out_put ;
  	6).trim 去掉空格或去掉前后匹配字符
		SELECT 
  			LENGTH(TRIM('    张翠上   ')) AS out_put ;

		SELECT TRIM('a' FROM 'aaaaaaaaaa张aaaaaaaaa翠aaaa山aaaaaaaa') AS out_put;
		SELECT TRIM('aa' FROM 'aaaaaaaaaaa张aaaaaaaaa翠aaaa山aaaaaaaaa') AS out_put;
    7).LPAD 用指定的字符实现左填充指定长度
    	SELECT 
  			LPAD('殷素素', 10, '*') AS out_put ;
		SELECT 
  			LPAD('殷素素', 2, '*') AS out_put ;
  	8).RPAD 用指定的字符实现右填充长度
  		SELECT 
 			RPAD('殷素素', 10, 'abcd') AS out_put ;
 	9).replace 替换
 		SELECT 
  			REPLACE (
    		'张无忌爱上了周芷若',
    		'周芷若',
    		'赵敏'
  			) AS out_put ;

2.数学函数
	1).round 四舍五入
		SELECT ROUND(1.45);
		SELECT ROUND(1.567, 2);
	2).ceil 向上取整,返回>=该参数的最小整数 SQL--CEILING
		SELECT CEIL(0.11);
		SELECT CEIL(1.00);
	3).floor 向下取整,返回<=该参数的最大整数
		SELECT FLOOR(-9.99);
	4).truncate 截断
		SELECT TRUNCATE (1.69999999,2);
	5).mod 取余
		mod(a,b) : a-a/b*b;
		SELECT MOD(-10, 3); -10 - -10 / 3 * 3
		SELECT MOD(10, -3);	10 - 10 /3 * 3
		
3.日期函数
	1)now 返回当前系统日期 + 时间
		SELECT NOW();
	2)curdate 返回当前系统日期,不包含时间
		SELECT CURDATE();
	3)curtime 返回当前时间
		SELECT CURTIME();
	4)可以获取指定的部分,年,月,日,小时,分钟,秒
		SELECT YEAR(NOW()) 年;
		SELECT YEAR('1999-1-1') 年;
		SELECT YEAR(hiredate) 年 FROM employees;
		SELECT MONTH(NOW()) 月;
		SELECT MONTHNAME(NOW()) 月;
		DAY
		HOUR
		MINUTE
		SECOND
	5)str_to_date
		SELECT 
  			STR_TO_DATE('1998-3-2', '%Y-%c-%d') AS out_put ;
		SELECT *
		FROM employees 
		WHERE hiredate = STR_TO_DATE('4-3 1992', '%c-%d %Y')
	6)date_fromat
		SELECT DATE_FORMAT(NOW(), '%y年%m月%d日') AS out_put;
		SELECT 
  			first_name,
  			DATE_FORMAT(hiredate, '%m月/%d日 %y年') AS 入职时间 
		FROM
  			employees 
		WHERE commission_pct IS NOT NULL ;
```

5.流程控制函数
	1)if函数 可以看作三目运算符
	SELECT IF('10 > 5', '大', '小');
	2)case函数 
	第一种用法类似与switch case
	case 要判断的字段或表达式
	when 常量1 then 要显示的值1<值后面不加分号>或语句1;<语句好像也不加分号>
	when 常量2 then 要显示的值2或语句2;
	...
	else 要显示的值?或语句?;
	end
	
	select salary 原始工资, department_id, #必须有,
	case department_id
	when 30 then salary * 1.1
	when 40 then salary * 1.2
	when 50 then salary * 1.3
	else salary
	end as 新工资
	from employees;
	
	第二种用法
	case
	when 条件1 then 要显示的值1或语句1
	when 条件2 then 要显示的值2或语句2
	...
	else 要显示的值?或语句?
	end
